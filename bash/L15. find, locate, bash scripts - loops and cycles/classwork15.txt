# поиск по названию
find /path/to/search -name "*.txt"
find /path/to/search -name "*.jpg"

# поиск по названию и размеру
find /path/to/search -name "*.txt" -size +1M

# поиск по дате изменения файла или директории
find /path/to/search -mtime -5

# find будет искать файлы внутри каталога /home/ и его подкаталогов, которые были изменены
# менее 5 дней назад, имеют расширение .txt и размер больше 15 килобайт
find /home/ -mtime -5 -name "*.txt" -size +15k

# Находит все обычные файлы в указанном каталоге и его подкаталогах, которые были модифицированы более 7 дней назад,
# и удаляет их
find /path/to/search -type f -mtime +7 -delete

# Находит все обычные файлы в указанном каталоге и его подкаталогах, а затем устанавливает права доступа 644 для
# каждого найденного файла
find /path/to/search -type f -exec chmod 644 {} \;

# find найдет все файлы с расширением .txt внутри каталога /root/ и его подкаталогов, а затем
# скопирует каждый найденный файл в каталог /home
find /root/ -type f -name "*.txt" -exec cp {} /home \;

  done
