# чтобы посмотреть коммиты на всех ветках, а не на текущей
git log --all --oneline
# для файлов, которые модифицированы, чтобы не делать лишний шаг с git add
git commit -a

# Отображение списка веток в репозитории
git branch
# Создание новой ветки с именем ＜ветка＞
git branch <branch>
# Удаление указанной ветки
git branch -d <branch>
# Принудительное удаление указанной ветки, даже если в ней есть неслитые изменения
git branch -D <branch>
# Изменение имени текущей ветки на ＜ветка＞
git branch -m <branch>
# Вывод списка всех удаленных веток
git branch -a
# просмотр списка всех веток в репозитории
git branch -v

# Переключение на другую ветку
git checkout <имя ветки>
# Создание и переключение на новую ветку
git checkout -b <имя ветки>
# Переключение на удаленную ветку
git checkout ＜remotebranch＞
# Создание новой ветки на основе удаленного репозитория
git checkout -b ＜remotebranch＞ origin/＜remotebranch＞
# Переключение на новую локальную ветку
git checkout -b ＜branchname＞
# Переключение на новую локальную ветку
git reset --hard origin/＜branchname＞
# откатить все изменённые файлы в рабочей директории к состоянию последнего коммита (то есть отменить все локальные
# изменения, которые ещё не были git add или git commit).
git checkout .

# Задание
Установка файла .gitignore:
1. Создать новый текстовый файл в корневом каталоге вашего Git репозитория.
2. Назвать файл .gitignore.
Пример : https://github.com/aliaskov/bashscripts/blob/master/.gitignore
3. Выполните команду, чтобы скачать файл .gitignore в текущий каталог:
curl -O https://raw.githubusercontent.com/aliaskov/bashscripts/master/.gitignore

# Создание глобального файла .gitignore для игнорирования файлов для всех репозиториев на вашей машине
git config --global core.excludesfile .gitignore

# клонирование удаленного репозитория
git clone

# полное копирование другого репозитория
git fork

# перемещение текущих изменений в локальную директорию
git stash save filename
git stash list
git stash apply stash_nummer
git stash pop stash_nummer

# Извлечение содержимого удаленной ветки
git fetch --all

# Восстановление удаленного файла, если удаление не было закоммичено
git restore password.txt

# Показывает, какие файлы будут удалены, но фактически ничего не удаляет. Только не закоммиченные файлы. Не трогает историю
git clean -xdn
# Это уже настоящее удаление файлов:
git clean -xdf

# Это инструмент для очистки истории, например при случайном коммите паролей, ключей, токенов и т.п. Проходит по каждому
# коммиту в истории, в каждом коммите удаляет файл passwords, переписывает коммиты, генерирует новую историю репозитория,
# старые коммиты (с паролями) становятся недоступными
git filter-branch --tree-filter 'rm -f passwords' HEAD
